// BHC 独立核心模块 - 简化版编译配置
// 仅编译核心加密模块用于测试

cmake_minimum_required(VERSION 3.16)
project(BHC_Core VERSION 1.0.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(MSVC)
    add_compile_options(/W4 /utf-8)
    add_definitions(-DWIN32_LEAN_AND_MEAN -D_CRT_SECURE_NO_WARNINGS)
else()
    add_compile_options(-Wall -Wextra)
endif()

add_library(bhc_crypto STATIC
    src/crypto/progpow.cpp
    src/crypto/progpow.h
    src/crypto/mldsa.cpp
    src/crypto/mldsa.h
)

target_include_directories(bhc_crypto PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${CMAKE_CURRENT_SOURCE_DIR}/src/crypto
)

add_executable(bhc_test_crypto
    src/test/crypto_tests.cpp
)
target_link_libraries(bhc_test_crypto bhc_crypto)

add_executable(bhc_genesis
    tools/genesis_tool.cpp
)
target_link_libraries(bhc_genesis bhc_crypto)

message(STATUS "")
message(STATUS "BHC Core Build Configuration:")
message(STATUS "  C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "  Build Type: ${CMAKE_BUILD_TYPE}")
message(STATUS "")
